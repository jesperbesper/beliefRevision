# beliefRevision

# The program requires sympy 

    pip install sympy

# To run the program:

    python cli.py


# Command to use in the Computer line interface: 

    [r] Revise the beliefs of the belief base
    [c] Clears the belief base
    [v] Displays all beliefs in the belief base
    [h] Displays the menu
    [q] Quits the program
